%%%% Preamble %%%%%

  %% Document Class and Margins %%
	\documentclass[10pt]{book}
	% Margins to support printing on a full sheet of paper with 3 hole punching %
	\usepackage[letterpaper,top=1in,bottom=1in,right=1in,left=1in]{geometry}

  %% Packages Used for This Document %%
	
	% Package for colored cells in tables %
	\usepackage[table]{xcolor}	% Must be above tikz or graphicx for precedence

	% Package for including graphics %
	\usepackage{graphicx}

	% Package to indent all first paragraphs after section headings %
	\usepackage{indentfirst}
	
	% Package for underline and strikethrough fonts %
	\usepackage[normalem]{ulem}
	
	% Package for including PDF documents %
	\usepackage{pdfpages}
	
	% Package for commenting large sections of text %
	\usepackage{comment}
	
	% Package for full features outlines %
	\usepackage{outlines}

	% Package for creating long, cross page tables %
	\usepackage{longtable}

	% Package for spanning multiple rows in tables %
	\usepackage{multirow}

	% Package for turning schedule to landscape %
	\usepackage{lscape}
	
	% Package for Biolinum/Libertine Font %
	\usepackage[T1]{fontenc}
	\usepackage{libertine}

	% This font provides the \ding characters %
	\usepackage{pifont}

	% Package for creating customizable section titles %
	\usepackage[explicit,pagestyles]{titlesec}
	
	% Package for header and footer creation %
	\usepackage{fancyhdr}
	\pagestyle{fancy}

  %% Font Changes %%
	% Sets base font as sans serif Biolinum (Libertine) font
	\renewcommand*\familydefault{\sfdefault}

  %% Color Definition %%
	\definecolor{Color1}{HTML}{3504bd}

  %% Header and Footer %%
	\renewcommand{\chaptermark}[1]{% 
	\markboth{\color{black}--Chapter \thechapter-- \\ \color{Color1}{\bfseries #1}}{}} 

	% Width of the line in the footer %
	\renewcommand{\footrulewidth}{.5pt}

	% Options for the header %
	\lhead{\itshape Document Name Here}
	%\chead{} %Commented out so nothing goes here
	\rhead{\leftmark} %Puts the chapter in all right header spaces

	% Options for the footer %
	\lfoot{Revision: \today}
	\cfoot{\thepage}
	\rfoot{{\bfseries My Company Name}}

  %% Macro \newcommand Changes %%
	% Sets the HRule command for the title page %
	\newcommand{\HRule}{\rule{\linewidth}{0.5mm}}

  %% Set Environment Changes %%
	% Left align longtable environments
	\setlength{\LTleft}{0pt}

  %% Section and Subsection Style Changes %%

   % Chapter styling %
	\titleformat{name=\chapter}[display]
	{\flushright\bfseries}
	{\LARGE\itshape Chapter \thechapter} % label
	{8pt}
	{\rule{\textwidth}{2pt} \\ \color{Color1}\Huge\bfseries#1}

   % Section styling %
	\titleformat{name=\section}[frame]
	{\normalfont\Large\bfseries}
	{\filright\small\enspace Section \thesection\enspace}
	{8pt}
	{\color{Color1}\Large\bfseries#1}

   % Subsection styling %
	\titleformat{name=\subsection}
	{\normalfont\large\bfseries}
	{\color{Color1}\thesubsection.}{0pt}{\color{Color1}\hspace{1em}#1}
	[{\titlerule[2pt]}]
	
   % Subsubsection styling %
	\titleformat{name=\subsubsection}
	{\bfseries}
	{}{0pt}{\color{Color1}\hspace{1em}-- \textit{#1} --}

%%%% Document Class %%%%

  %% Document Beginning %%
	\begin{document}

  %% Title Page %%
	\input{title.tex}

  %% Table of Contents %%
	\setcounter{tocdepth}{2}
	\renewcommand\contentsname{Table of Contents}
	\tableofcontents
	
  %% Includes %%
	\input{toc.tex}				%This file contains all of the production chapters and \includes{}

  %% Document Ending %%
	\end{document}
